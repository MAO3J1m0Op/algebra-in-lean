\begin{definition}
    \label{definition : Homomorphism}
    \lean{AlgebraInLean.Group.Homomorphism}
    \leanok
    Of a map \( \phi \) mapping from a group \( G \) to a group \( H \), for
    elements \( g \in G \) and \( h \in H \), \( \phi \) is a homomorphism if and only if:
    \[
        \phi(g)\phi(h) = \phi(gh)
    \]
\end{definition}

\end{proof}
