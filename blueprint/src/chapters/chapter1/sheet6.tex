\section{$D_n$ as a group}

\begin{definition}[Dihedral Group]
    \label{definition : Dihedral}
    \lean{AlgebraInLean.Dihedral}
    \leanok
    \uses{definition : Group}
    $D_n$ is the dihedral group of order $2n$. This is the group of all
    symmetries of a regular n-gon. This includes n rotations and n reflections.
    It is represented by a pair of a boolean and an element of \verb|Fin n|.
    The boolean represents whether the element is a rotation or not, and the
    element of \verb|Fin n| represents how much the first vertex is rotated.
    The identity is a $0^{\circ}$ rotation, the inverse of an $x^{\circ}$
    rotation is a $(360 - x)^{\circ}$ rotation, and any reflection is its own
    inverse.
\end{definition}

\begin{theorem}
    \label{theorem : D₃_nonabelian}
    \lean{AlgebraInLean.D₃_nonabelian}
    \leanok
    \uses{definition : Dihedral}
    $D_3$ is the first example we've seen of a nonabelian group.
\end{theorem}
